Code example

Filtering the items array according to the values of the filter array - activeFilterParams

            
const activeFilterParams = {
    color: 'red',
    size: 'm',
  };
  const items = [
    {
      id: 1,
      color: 'red',
      size: 'm',
      sex: 'man',
    },
    {
      id: 2,
      color: 'blue',
      size: 'm',
      sex: 'woman',
    },    
  ];

  const filter = (items, params) => {
    if (items.length === 0) {
      return items;
    }
    const paramKeys = Object.keys(params);
    return items.filter((item) => {
      return paramKeys.every((paramKey) => item[paramKey] === params[paramKey]);
    })
  };

  filter(items, activeFilterParams);
